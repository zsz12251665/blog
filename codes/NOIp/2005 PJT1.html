Solution

Well, we should ignore the bench and make Tao Tao 30 centimetres higher. Then count the number of the apples which she is able to pick.

Code

Pascal

var
    apple:array[1..10]of integer;
    h,i,ans:integer;
begin
    for i:=1 to 10 do//Get the apples
        read(apple[i]);
    read(h);
    ans:=0;
    for i:=1 to 10 do//Count the apples
        if h+30>=apple[i] then
            inc(ans);
    write(ans);
end.

C++

#include<cstdio>
int apple[10];
int main()
{
    int h,ans=0;
    for(int i=0;i<10;i++)//Get the apples
        scanf("%d",&apple[i]);
    scanf("%d",&h);
    for(int i=0;i<10;i++)//Count the apples
        if(h+30>=apple[i])
            ans++;
    printf("%d",ans);
    return 0;
}
